## 四向碰撞

**正面**: 33个Skill分散在10个层级中，每个Skill独立工作。但人类遇到复杂问题时不会先想"我需要用哪个Skill"——他们自然地调用多种认知能力。ultimate-cognitive-orchestrator就是让Agent像人类一样：**收到问题→自动调度所有相关能力→综合输出**。这是33个Skill的"统一内核"。

**反面**: 统一调度33个Skill不能是简单串行（太慢）或简单并行（噪音太多）。必须是**层次化调度**：先快速识别任务类型→选择调度模式(单Skill/多Skill协同/全Skill涌现)→执行→自进化。需要一个"调度内核"像操作系统调度进程一样调度Skill。

**侧面**: 操作系统调度进程时有调度算法（Round-Robin/优先级/CFS）。认知调度也应该有算法：基于任务复杂度、领域相关性、历史成功率、Skill间依赖关系的多因子调度。关键洞察：**Skill之间有依赖DAG**——meta-skill必须在occupation-skill之前加载，fusion-skill需要两个domain-skill作为输入。

**整体**: 这是整个Skill工厂的"终极节点"——它不替代任何现有Skill，而是**所有Skill的统一入口和调度器**。任何用户问题只需要这一个接口，它会自动路由到正确的Skill组合。同时它自进化——每次调度都产生经验数据，反哺框架进化。

## 核心伪代码

```python
from __future__ import annotations
from dataclasses import dataclass, field
from typing import Optional
from enum import Enum


class ScheduleMode(Enum):
    SINGLE_SKILL = "single"           # 单Skill直接执行
    MULTI_SKILL_CHAIN = "chain"       # 多Skill链式执行
    MULTI_SKILL_PARALLEL = "parallel" # 多Skill并行执行
    FULL_ORCHESTRATION = "full"       # 全流程编排(DAG调度)
    EMERGENT = "emergent"             # 认知涌现模式


@dataclass
class SchedulePlan:
    """调度计划"""
    mode: ScheduleMode
    skills_to_load: list[str]          # 按依赖顺序
    domains_active: list[str]          # 活跃领域
    dimensions_loaded: list[str]       # 加载的维度
    estimated_confidence: float        # 预估置信度
    execution_order: list[str]         # DAG拓扑排序结果


class UltimateCognitiveOrchestrator:
    """终极认知编排器 — 33个Skill的统一调度内核"""

    def __init__(self):
        self.runtime = DomainOrchestratorRuntime()      # 运行时调度
        self.engine = SelfEvolvingDomainEngine()          # 框架自进化
        self.dag = SkillDependencyDAG(SKILL_DAG)          # Skill依赖图
        self.history = ScheduleHistory("data/schedule_history.jsonl")
        self.identity = ExpertIdentity("expert-identity.md")  # 144节点

    def think(self, task: str, context: list[str] = None) -> OrchestratedOutput:
        """
        主入口: 任何问题→统一调度→综合输出
        
        这是终极认知编排器的唯一公开接口。
        用户不需要知道33个Skill的存在——只需要调用think()。
        """
        # ═══ Phase 1: 任务理解 ═══
        task_profile = self._profile_task(task, context)
        # task_profile包含: domains, complexity, type, urgency

        # ═══ Phase 2: 调度规划 ═══
        plan = self._create_schedule_plan(task_profile)
        # 根据复杂度选择调度模式:
        #   complexity=low  → SINGLE_SKILL (1个Skill)
        #   complexity=mid  → MULTI_SKILL_CHAIN (2-3个Skill)
        #   complexity=high → FULL_ORCHESTRATION (DAG调度)
        #   complexity=max  → EMERGENT (全Skill涌现)

        # ═══ Phase 3: 依赖加载 ═══
        loaded_skills = self._load_with_dependencies(plan.skills_to_load)
        # 按DAG拓扑排序加载，确保依赖先于使用

        # ═══ Phase 4: 执行调度 ═══
        if plan.mode == ScheduleMode.SINGLE_SKILL:
            result = self._execute_single(loaded_skills[0], task, task_profile)
        elif plan.mode == ScheduleMode.MULTI_SKILL_CHAIN:
            result = self._execute_chain(loaded_skills, task, task_profile)
        elif plan.mode == ScheduleMode.MULTI_SKILL_PARALLEL:
            result = self._execute_parallel(loaded_skills, task, task_profile)
        elif plan.mode == ScheduleMode.FULL_ORCHESTRATION:
            result = self._execute_dag(loaded_skills, task, task_profile)
        else:  # EMERGENT
            result = self._execute_emergent(task, task_profile)

        # ═══ Phase 5: 综合输出 ═══
        output = self._synthesize_output(result, plan, task_profile)

        # ═══ Phase 6: 自进化 ═══
        self._record_and_evolve(task, plan, result, output)

        return output

    # ─── Phase 1: 任务画像 ───

    def _profile_task(self, task: str, context: list[str]) -> TaskProfile:
        """任务画像: 领域+复杂度+类型+紧急度"""
        # 复用domain-orchestrator-runtime的领域识别
        domain_scores = self.runtime.detect_domains(task, context)
        
        # 复杂度评估
        primary_domains = [ds for ds in domain_scores if ds.score >= 0.5]
        if len(primary_domains) <= 1:
            complexity = "low"
        elif len(primary_domains) <= 3:
            complexity = "mid"
        elif len(primary_domains) <= 6:
            complexity = "high"
        else:
            complexity = "max"
        
        # 任务类型
        task_type = self._classify_task_type(task)
        # decision/verification/learning/creation/execution
        
        return TaskProfile(
            domains=domain_scores,
            primary=primary_domains,
            complexity=complexity,
            type=task_type,
            urgency=self._assess_urgency(task),
        )

    # ─── Phase 2: 调度规划 ───

    def _create_schedule_plan(self, profile: TaskProfile) -> SchedulePlan:
        """根据任务画像选择调度模式"""
        COMPLEXITY_MODE = {
            "low":  ScheduleMode.SINGLE_SKILL,
            "mid":  ScheduleMode.MULTI_SKILL_CHAIN,
            "high": ScheduleMode.FULL_ORCHESTRATION,
            "max":  ScheduleMode.EMERGENT,
        }
        mode = COMPLEXITY_MODE[profile.complexity]
        
        # 选择需要加载的Skill
        if mode == ScheduleMode.SINGLE_SKILL:
            skills = self._select_single_skill(profile)
        elif mode == ScheduleMode.MULTI_SKILL_CHAIN:
            skills = self._select_chain_skills(profile)
        else:
            skills = self._select_all_relevant_skills(profile)
        
        # DAG拓扑排序
        ordered = self.dag.topological_sort(skills)
        
        return SchedulePlan(
            mode=mode,
            skills_to_load=ordered,
            domains_active=[d.domain for d in profile.primary],
            dimensions_loaded=self._compute_needed_dims(profile),
            estimated_confidence=self._estimate_confidence(profile, mode),
            execution_order=ordered,
        )

    # ─── Phase 3: 依赖加载 ───

    def _load_with_dependencies(self, skills: list[str]) -> list[SkillInstance]:
        """按DAG拓扑排序加载Skill及所有依赖"""
        loaded = []
        for skill_name in skills:
            deps = self.dag.get_dependencies(skill_name)
            for dep in deps:
                if dep not in [s.name for s in loaded]:
                    loaded.append(self._instantiate_skill(dep))
            loaded.append(self._instantiate_skill(skill_name))
        return loaded

    # ─── Phase 4: 执行 ───

    def _execute_single(self, skill, task, profile) -> SkillResult:
        """单Skill直接执行"""
        return skill.execute(task, profile)

    def _execute_chain(self, skills, task, profile) -> ChainedResult:
        """链式执行: Skill A的输出→Skill B的输入"""
        results = []
        context = task
        for skill in skills:
            result = skill.execute(context, profile)
            results.append(result)
            context = f"{task}\n\n前序分析结果:\n{result.summary}"
        return ChainedResult(results=results, final_context=context)

    def _execute_parallel(self, skills, task, profile) -> ParallelResult:
        """并行执行多个Skill，结果合并"""
        results = [skill.execute(task, profile) for skill in skills]
        return ParallelResult(results=results)

    def _execute_dag(self, skills, task, profile) -> DAGResult:
        """DAG调度: 按依赖关系并行执行"""
        layers = self.dag.get_execution_layers(skills)
        all_results = {}
        for layer in layers:
            # 同层Skill并行执行
            for skill_name in layer:
                deps_results = {d: all_results[d] for d in self.dag.get_dependencies(skill_name) if d in all_results}
                skill = self._instantiate_skill(skill_name)
                result = skill.execute(task, profile, deps_results)
                all_results[skill_name] = result
        return DAGResult(results=all_results)

    def _execute_emergent(self, task, profile) -> EmergentResult:
        """认知涌现: 全Skill参与，检测涌现"""
        # 加载所有相关Skill
        all_skills = self._select_all_relevant_skills(profile)
        loaded = self._load_with_dependencies(all_skills)
        
        # 全部执行
        all_results = {}
        for skill in loaded:
            all_results[skill.name] = skill.execute(task, profile, all_results)
        
        # 认知涌现检测
        emergent_points = self._detect_emergence(all_results, task)
        
        return EmergentResult(results=all_results, emergent=emergent_points)

    # ─── Phase 5: 综合输出 ───

    def _synthesize_output(self, result, plan, profile) -> OrchestratedOutput:
        """综合所有Skill的输出为最终答案"""
        # 收集所有认知点
        all_points = self._collect_cognitive_points(result)
        
        # 按领域和维度组织
        organized = self._organize_by_domain_dimension(all_points, profile)
        
        # 置信度综合
        overall_confidence = self._compute_overall_confidence(all_points)
        
        # 生成结构化答案
        answer = self._format_answer(organized, profile)
        
        # 生成警告
        warnings = self._generate_warnings(all_points, overall_confidence, profile)
        
        return OrchestratedOutput(
            answer=answer,
            confidence=overall_confidence,
            domains_used=plan.domains_active,
            skills_used=[s for s in plan.skills_to_load],
            dimensions_used=plan.dimensions_loaded,
            cognitive_points=all_points,
            warnings=warnings,
            schedule_mode=plan.mode,
        )

    # ─── Phase 6: 自进化 ───

    def _record_and_evolve(self, task, plan, result, output):
        """记录调度经验 → 触发框架自进化"""
        # 记录到历史
        self.history.record({
            "task": task[:200],
            "mode": plan.mode.value,
            "skills": plan.skills_to_load,
            "confidence": output.confidence,
            "timestamp": now_iso(),
        })
        
        # 检查是否需要进化
        if output.confidence < 0.85:
            # 置信度低 → 可能缺少维度/Skill
            self.engine.signal_evolution(f"low_confidence: {output.confidence:.2f}")
        
        if self.history.count_recent(hours=1) > 10:
            # 高频使用 → 检查是否有新领域趋势
            self.engine.run_evolution_cycle()

@dataclass
class TaskProfile:

    primary: list
    complexity: str  # low/mid/high/max
    type: str        # decision/verification/learning/creation/execution
    urgency: str     # normal/high/critical

@dataclass
class OrchestratedOutput:
    answer: str
    confidence: float
    domains_used: list[str]
    skills_used: list[str]
    dimensions_used: list[str]
    cognitive_points: list
    warnings: list
    schedule_mode: ScheduleMode
```
