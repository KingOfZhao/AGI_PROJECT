@dataclass
class Interval:
    """代表一个数值区间 [min, max]，支持区间运算"""
    min_val: float
    max_val: float

    def __post_init__(self):
        if self.min_val > self.max_val:
            # 允许反向区间初始化，但在运算时视为无效或空集
            logger.warning(f"Invalid interval created: [{self.min_val}, {self.max_val}]")

    def is_valid(self) -> bool:
        return self.min_val <= self.max_val

    def intersect(self, other: 'Interval') -> 'Interval':
        """区间交集 (约束收紧)"""
        return Interval(max(self.min_val, other.min_val), min(self.max_val, other.max_val))

    def scale(self, factor: float) -> 'Interval':
        """区间缩放 (物理公式映射)"""
        v1, v2 = self.min_val * factor, self.max_val * factor
        return Interval(min(v1, v2), max(v1, v2))

    def add_margin(self, abs_margin: float) -> 'Interval':
        """增加绝对误差裕度"""
        return Interval(self.min_val - abs_margin, self.max_val + abs_margin)

    def to_dict(self):
        return {"min": round(self.min_val, 4), "max": round(self.max_val, 4)}

class FVFSolverChain:
    """
    F→V→F 约束传播求解器
    实现链式收敛: F0 -> V1 -> F1 -> V2 -> ...
    """
    def __init__(self):
        self.chain: List[ConstraintNode] = []
        self.context: Dict[str, Interval] = {} # 存储所有节点的当前值

    def add_node(self, node: ConstraintNode):
        self.chain.append(node)
        if node.value:
            self.context[node.id] = node.value

    def propagate(self):
        logger.info("--- 开始 F→V→F 链式传播 ---")
        
        # 简单的顺序传播 (实际复杂系统可能需要多次迭代直至收敛)
        # 链路: F0 -> V1 -> F1 -> V2 ...
        
        for i in range(len(self.chain) - 1):
            current_node = self.chain[i]
            next_node = self.chain[i+1]
            
            logger.info(f"传播步骤 {i}: {current_node.id} ({current_node.type.value}) -> {next_node.id} ({next_node.type.value})")

            if not current_node.value or not current_node.value.is_valid():
                logger.error(f"阻塞: 节点 {current_node.id} 值无效，传播终止。")
                break

            # 1. F -> V: 固定约束限制变量空间 (直接赋值或取交集)
            # 2. V -> F: 变量收敛后固化为新的约束 (计算并固化)
            
            if current_node.type == NodeType.F_FIXED and next_node.type == NodeType.V_VARIABLE:
                # F -> V: 下游 V 的取值范围不能超出上游 F 的限制
                # 这里假设 F 的 constraint_fn 定义了如何限制 V
                if current_node.constraint_fn:
                    allowed_range = current_node.constraint_fn(self.context)
                    if next_node.value:
                        # 取交集 (初始假设 vs 物理约束)
                        next_node.value = next_node.value.intersect(allowed_range)
                    else:
                        next_node.value = allowed_range
                    
                    logger.info(f"  F→V 收敛: {next_node.name} 初始范围受限更新为 {next_node.value.to_dict()}")
                else:
                    # 默认直接传递
                    next_node.value = current_node.value 

            elif current_node.type == NodeType.V_VARIABLE and next_node.type == NodeType.F_FIXED:
                # V -> F: 使用 V 的值通过物理公式计算出 F
                if next_node.constraint_fn:
                    calculated_val = next_node.constraint_fn(self.context)
                    # 固化: 将计算结果赋值给 F (或与 F 原有默认值取交集)
                    next_node.value = calculated_val 
                    logger.info(f"  V→F 固化: {next_node.name} 计算得出 {next_node.value.to_dict()}")
            
            # 更新上下文
            self.context[next_node.id] = next_node.value

        logger.info("--- 传播结束 ---")
        return self.chain

    def export_json(self) -> str:
        """导出为 deduction_export.json 格式"""
        results = []
        for node in self.chain:
            if node.value:
                results.append({
                    "step": node.id,
                    "type": node.type.value,
                    "name": node.name,
                    "result_interval": node.value.to_dict(),
                    "timestamp": datetime.now().isoformat()
                })
        return json.dumps({"fv_chain_results": results}, indent=2, ensure_ascii=False)

@dataclass
class Interval:
    """代表一个数值区间 [min, max]，支持区间运算"""
    min_val: float
    max_val: float

    def __post_init__(self):
        if self.min_val > self.max_val:
            # 允许反向区间初始化，但在运算时视为无效或空集
            logger.warning(f"Invalid interval created: [{self.min_val}, {self.max_val}]")

    def is_valid(self) -> bool:
        return self.min_val <= self.max_val

    def intersect(self, other: 'Interval') -> 'Interval':
        """区间交集 (约束收紧)"""
        return Interval(max(self.min_val, other.min_val), min(self.max_val, other.max_val))

    def scale(self, factor: float) -> 'Interval':
        """区间缩放 (物理公式映射)"""
        v1, v2 = self.min_val * factor, self.max_val * factor
        return Interval(min(v1, v2), max(v1, v2))

    def add_margin(self, abs_margin: float) -> 'Interval':
        """增加绝对误差裕度"""
        return Interval(self.min_val - abs_margin, self.max_val + abs_margin)

    def to_dict(self):
        return {"min": round(self.min_val, 4), "max": round(self.max_val, 4)}

            current_node = self.chain[i]
            next_node = self.chain[i+1]
            
            logger.info(f"传播步骤 {i}: {current_node.id} ({current_node.type.value}) -> {next_node.id} ({next_node.type.value})")

    min_val: float
    max_val: float

    def __post_init__(self):
        if self.min_val > self.max_val:
            # 允许反向区间初始化，但在运算时视为无效或空集
            logger.warning(f"Invalid interval created: [{self.min_val}, {self.max_val}]")

        return self.min_val <= self.max_val

        v1, v2 = self.min_val * factor, self.max_val * factor
        return Interval(min(v1, v2), max(v1, v2))

        """增加绝对误差裕度"""

        logger.info("--- 开始 F→V→F 链式传播 ---")

        # 链路: F0 -> V1 -> F1 -> V2 ...
        
        for i in range(len(self.chain) - 1):
            current_node = self.chain[i]
            next_node = self.chain[i+1]
            
            logger.info(f"传播步骤 {i}: {current_node.id} ({current_node.type.value}) -> {next_node.id} ({next_node.type.value})")

            if not current_node.value or not current_node.value.is_valid():
                logger.error(f"阻塞: 节点 {current_node.id} 值无效，传播终止。")
                break

            if current_node.type == NodeType.F_FIXED and next_node.type == NodeType.V_VARIABLE:

                else:

            elif current_node.type == NodeType.V_VARIABLE and next_node.type == NodeType.F_FIXED:
