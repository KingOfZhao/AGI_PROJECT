*   **法律红线**：创作者分成超过 800 元/次触发劳务报酬所得税预扣预缴，这是物理般的法定事实。
*   **业务本质**：平台的核心是“价值流转”（内容/服务 -> 用户付费 -> 平台/创作者分账）。

**双轨策略推演**

    *   设计状态机：`PENDING` (待结算) -> `COMPLIANCE_CHECK` (合规审查/税务申报) -> `SETTLED` (已结算)。

*   **[BLOCKED] 支付牌照/二清合规**：代码层面无法解决金融资质问题。必须接入第三方聚合支付（如 Ping++, Stripe）或银行存管系统，需法务/商务介入。
*   **[BLOCKED] 具体税率计算**：劳务报酬所得税率阶梯（20%-40%）需税务顾问确认具体计算公式（是否含增值税附加等）。

"""
Project: p_rose (予人玫瑰)
Module: CRM & Compliance Core
Framework: FastAPI + SQLAlchemy (Async)
Description: 实现CRM基础功能及基于“800元税务红线”的合规资金拦截框架。
"""

import datetime
from typing import List, Optional
from enum import Enum
from decimal import Decimal

from fastapi import FastAPI, Depends, HTTPException, status
from fastapi.security import OAuth2PasswordBearer, OAuth2PasswordRequestForm
from sqlalchemy import (
    create_engine, Column, Integer, String, Boolean, 
    DateTime, Numeric, Enum as SQLEnum, ForeignKey
)
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.future import select
from sqlalchemy.orm import sessionmaker, Session, relationship
from pydantic import BaseModel, EmailStr, confloat
from passlib.context import CryptContext

# ================= 配置与常量 =================

# 模拟数据库配置
SQLALCHEMY_DATABASE_URL = "sqlite:///./p_rose.db"
# SQLALCHEMY_DATABASE_URL = "postgresql://user:pass@localhost/db" # 生产环境替换

# 安全配置
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")
oauth2_scheme = OAuth2PasswordBearer(tokenUrl="token")

# 核心合规常量 (已知约束 F)
TAX_THRESHOLD_CNY = Decimal("800.00") # 劳务报酬所得税起征点

app = FastAPI(title="p_rose CRM Engine", version="0.1.0")

# ================= 数据库模型 =================

Base = declarative_base()

class UserRole(str, Enum):
    ADMIN = "admin"
    CREATOR = "creator"
    USER = "user"

class TaskStatus(str, Enum):
    OPEN = "open"
    IN_PROGRESS = "in_progress"
    COMPLETED = "completed"

class PayoutStatus(str, Enum):
    PENDING = "pending"             # 待结算
    AUTO_SETTLE = "auto_settle"     # 自动打款 (<800)
    COMPLIANCE_REVIEW = "compliance" # 合规审查/需报税 (>=800)
    SETTLED = "settled"             # 已完成
    FAILED = "failed"

class User(Base):
    __tablename__ = "users"
    id = Column(Integer, primary_key=True, index=True)
    email = Column(String, unique=True, index=True)
    hashed_password = Column(String)
    role = Column(String, default=UserRole.USER)
    is_active = Column(Boolean, default=True)
    
    # 关系
    tasks_created = relationship("Task", back_populates="owner", foreign_keys="Task.owner_id")
    tasks_assigned = relationship("Task", back_populates="assignee", foreign_keys="Task.assignee_id")

class Task(Base):
    __tablename__ = "tasks"
    id = Column(Integer, primary_key=True, index=True)
    title = Column(String, index=True)
    description = Column(String)
    reward_amount = Column(Numeric(10, 2), default=0.00) # 悬赏金额
    status = Column(String, default=TaskStatus.OPEN)
    owner_id = Column(Integer, ForeignKey("users.id"))
    assignee_id = Column(Integer, ForeignKey("users.id"), nullable=True)
    
    owner = relationship("User", back_populates="tasks_created", foreign_keys=[owner_id])
    assignee = relationship("User", back_populates="tasks_assigned", foreign_keys=[assignee_id])

class PayoutRecord(Base):
    __tablename__ = "payouts"
    id = Column(Integer, primary_key=True, index=True)
    user_id = Column(Integer, ForeignKey("users.id"))
    amount = Column(Numeric(10, 2))
    status = Column(String, default=PayoutStatus.PENDING)
    created_at = Column(DateTime, default=datetime.datetime.utcnow)
    processed_at = Column(DateTime, nullable=True)
    log = Column(String, nullable=True) # 记录处理日志

# 数据库连接
engine = create_engine(SQLALCHEMY_DATABASE_URL, connect_args={"check_same_thread": False})
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

Base.metadata.create_all(bind=engine)

# ================= Pydantic Schemas =================

class UserBase(BaseModel):
    email: EmailStr

class UserCreate(UserBase):
    password: str

class UserResponse(UserBase):
    id: int
    role: str
    class Config:
        orm_mode = True

class TaskBase(BaseModel):
    title: str
    description: Optional[str] = None
    reward_amount: confloat(ge=0.0) = 0.0

class TaskCreate(TaskBase):
    pass

class TaskResponse(TaskBase):
    id: int
    status: str
    owner_id: int
    class Config:
        orm_mode = True

# ================= 依赖与工具 =================

def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

def verify_password(plain_password, hashed_password):
    return pwd_context.verify(plain_password, hashed_password)

def get_password_hash(password):
    return pwd_context.hash(password)

def get_current_user(token: str = Depends(oauth2_scheme), db: Session = Depends(get_db)):
    # 简化版认证，生产环境需使用 JWT 解码
    user = db.query(User).filter(User.email == token).first() # Mock: token is email for simplicity
    if not user:
        raise HTTPException(status_code=401, detail="Invalid credentials")
    return user

# ================= 核心业务逻辑 (推演重点) =================

def process_payout(db: Session, user_id: int, amount: Decimal):
    """
    资金流合规自解框架实现
    目的：处理创作者分成，自动处理税务合规边界
    """
    new_payout = PayoutRecord(user_id=user_id, amount=amount)
    
    # 1. 边界条件检查 (F -> 800元红线)
    if amount < TAX_THRESHOLD_CNY:
        # 已知路径：低于阈值，走自动结算
        new_payout.status = PayoutStatus.AUTO_SETTLE
        new_payout.log = f"Amount {amount} < {TAX_THRESHOLD_CNY}. Auto-settlement triggered."
        # 这里接入支付宝/微信企业付款API
    else:
        # 未知路径：超过阈值，进入合规队列
        # 不尝试解决具体的税务计算，而是将问题“悬置”并通知人工/税务模块
        new_payout.status = PayoutStatus.COMPLIANCE_REVIEW
        new_payout.log = f"Amount {amount} >= {TAX_THRESHOLD_CNY}. Requires tax withholding. Moved to Compliance Review."
        # TODO: 触发通知系统，告知管理员需处理税务申报
    
    db.add(new_payout)
    db.commit()
    db.refresh(new_payout)
    return new_payout

@app.post("/token")
async def login(form_data: OAuth2PasswordRequestForm = Depends(), db: Session = Depends(get_db)):
    # 简化登录逻辑：如果用户不存在则创建（仅用于Demo）
    user = db.query(User).filter(User.email == form_data.username).first()
    if not user:
        hashed_pw = get_password_hash(form_data.password)
        user = User(email=form_data.username, hashed_password=hashed_pw)
        db.add(user)
        db.commit()
    elif not verify_password(form_data.password, user.hashed_password):
        raise HTTPException(status_code=400, detail="Incorrect password")
    
    return {"access_token": user.email, "token_type": "bearer"} # Mock Token

@app.post("/tasks/", response_model=TaskResponse)
def create_task(task: TaskCreate, current_user: User = Depends(get_current_user), db: Session = Depends(get_db)):
    """创建任务（付费点探索）"""
    db_task = Task(**task.dict(), owner_id=current_user.id)
    db.add(db_task)
    db.commit()
    db.refresh(db_task)
    return db_task

@app.post("/tasks/{task_id}/complete")
def complete_task(task_id: int, current_user: User = Depends(get_current_user), db: Session = Depends(get_db)):
    """完成任务并触发资金流"""
    task = db.query(Task).filter(Task.id == task_id).first()
    if not task:
        raise HTTPException(status_code=404, detail="Task not found")
    
    # 模拟分配任务给当前用户
    task.assignee_id = current_user.id
    task.status = TaskStatus.COMPLETED
    
    # 触发结算逻辑 (关键推演步骤)
    payout_record = process_payout(db, current_user.id, Decimal(str(task.reward_amount)))
    
    return {
        "message": "Task completed", 
        "payout_status": payout_record.status, 
        "note": payout_record.log
    }

@app.get("/admin/pending_payouts/", response_model=List[dict])
def view_pending_payouts(current_user: User = Depends(get_current_user), db: Session = Depends(get_db)):
    """管理员视图：查看待处理的税务合规订单"""
    if current_user.role != UserRole.ADMIN:
        raise HTTPException(status_code=403, detail="Admin only")
    
    pending = db.query(PayoutRecord).filter(
        PayoutRecord.status == PayoutStatus.COMPLIANCE_REVIEW
    ).all()
    
    return [{
        "id": p.id, 
        "user_id": p.user_id, 
        "amount": float(p.amount), 
        "status": p.status,
        "action_required": "Manual tax withholding required"
    } for p in pending]

def process_payout(db: Session, user_id: int, amount: Decimal):
    """
    资金流合规自解框架实现

        # 已知路径：低于阈值，走自动结算

        new_payout.log = f"Amount {amount} < {TAX_THRESHOLD_CNY}. Auto-settlement triggered."

        # 未知路径：超过阈值，进入合规队列

    db.add(new_payout)
    db.commit()
    db.refresh(new_payout)
    return new_payout

def process_payout(db: Session, user_id: int, amount: Decimal):
    """
    资金流合规自解框架实现
    目的：处理创作者分成，自动处理税务合规边界
    """
    new_payout = PayoutRecord(user_id=user_id, amount=amount)
    
    # 1. 边界条件检查 (F -> 800元红线)
    if amount < TAX_THRESHOLD_CNY:
        # 已知路径：低于阈值，走自动结算
        new_payout.status = PayoutStatus.AUTO_SETTLE
        new_payout.log = f"Amount {amount} < {TAX_THRESHOLD_CNY}. Auto-settlement triggered."
        # 这里接入支付宝/微信企业付款API
    else:
        # 未知路径：超过阈值，进入合规队列
        # 不尝试解决具体的税务计算，而是将问题“悬置”并通知人工/税务模块
        new_payout.status = PayoutStatus.COMPLIANCE_REVIEW
        new_payout.log = f"Amount {amount} >= {TAX_THRESHOLD_CNY}. Requires tax withholding. Moved to Compliance Review."
        # TODO: 触发通知系统，告知管理员需处理税务申报
    
    db.add(new_payout)
    db.commit()
    db.refresh(new_payout)
    return new_payout

@app.post("/token")
async def login(form_data: OAuth2PasswordRequestForm = Depends(), db: Session = Depends(get_db)):
    # 简化登录逻辑：如果用户不存在则创建（仅用于Demo）
    user = db.query(User).filter(User.email == form_data.username).first()
    if not user:
        hashed_pw = get_password_hash(form_data.password)
        user = User(email=form_data.username, hashed_password=hashed_pw)
        db.add(user)
        db.commit()
    elif not verify_password(form_data.password, user.hashed_password):
        raise HTTPException(status_code=400, detail="Incorrect password")
    
    return {"access_token": user.email, "token_type": "bearer"} # Mock Token

@app.post("/tasks/{task_id}/complete")
def complete_task(task_id: int, current_user: User = Depends(get_current_user), db: Session = Depends(get_db)):
    """完成任务并触发资金流"""
    task = db.query(Task).filter(Task.id == task_id).first()
    if not task:
        raise HTTPException(status_code=404, detail="Task not found")
    
    # 模拟分配任务给当前用户
    task.assignee_id = current_user.id
    task.status = TaskStatus.COMPLETED
    
    # 触发结算逻辑 (关键推演步骤)
    payout_record = process_payout(db, current_user.id, Decimal(str(task.reward_amount)))
    
    return {
        "message": "Task completed", 
        "payout_status": payout_record.status, 
        "note": payout_record.log
    }
