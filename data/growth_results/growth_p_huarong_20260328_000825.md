此问题的本质是**在固定网格(6mm)内设计一套自锁拼接系统**。活字印刷的"铅块独立铸造→排列使用→回收复用"模式可直接映射，但华容道式拼接引入了**平面内互锁**需求，这是与传统活字（仅靠字架外框固定）的关键差异。结构设计的核心矛盾是：**模块越小越灵活，但连接强度和定位精度越难保证**。

class MaterialType(Enum):
    PLA = {"name": "PLA", "temp_limit": 60, "shrinkage": 0.005}  # 收缩率0.5%
    PETG = {"name": "PETG", "temp_limit": 75, "shrinkage": 0.008}
    ABS = {"name": "ABS", "temp_limit": 100, "shrinkage": 0.012} # 收缩率1.2%

@dataclass
class DieModuleSpec:
    """
    刀模单体模块规格 (基于6mm网格)
    Core Logic: 
    1. 基础尺寸 6x6mm (Industry Standard)
    2. 互锁结构: 十字滑块 实现华容道式拼接
    3. 刀槽: 居中
    """
    module_id: str
    grid_unit: float = 6.0       # IADD通用最小网格
    height: float = 23.8         # 标准刀高 (未含底座)
    base_thickness: float = 5.0  # 底座厚度
    
    # 互锁参数
    tolerance: float = 0.15      # 打印公差
    lock_depth: float = 0.8      # 卡扣深度
    lock_width: float = 1.0      # 卡扣宽度
    
    # 刀片参数
    blade_thickness: float = 0.45 # 标准刀厚
    blade_width: float = 0.71    # 重型刀厚
    
    material: MaterialType = MaterialType.PLA

    def __post_init__(self):
        # 根据材料收缩率调整设计尺寸
        shrinkage = self.material.value['shrinkage']
        self.nominal_size = self.grid_unit * (1 + shrinkage)

class DieModuleGenerator:
    """
    生成模块化刀模的2D图纸 (SVG)
    解决问题: 
    - 6mm尺寸下的连接强度
    - PLA高温形变的公差补偿
    """
    
    def __init__(self, spec: DieModuleSpec):
        self.spec = spec
        self.svg_content = []
        
    def _generate_header(self) -> str:
        # 拓竹P2S 打印尺寸 256x256
        return f"""<svg width="256mm" height="256mm" viewBox="0 0 256 256" 
                xmlns="http://www.w3.org/2000/svg">
                <rect width="100%" height="100%" fill="#f0f0f0"/>
                <text x="10" y="20" font-family="Arial" font-size="5" fill="black">
                [P_Huarong] Module Layout | Date: {datetime.datetime.now().strftime('%Y-%m-%d')}
                </text>
                <text x="10" y="28" font-family="Arial" font-size="4" fill="gray">
                Material: {self.spec.material.value['name']} | Tolerance: {self.spec.tolerance}mm
                </text>
                <g transform="translate(20, 40)">
                """

    def _generate_module_path(self, x_offset: float, y_offset: float) -> str:
        """
        生成单个模块的2D投影路径
        结构: 
        - 外轮廓 (6x6mm)
        - 侧面锁扣 (Left/Right Slide, Top/Bottom Slide)
        - 中心刀槽
        """
        S = self.spec.nominal_size
        T = self.spec.tolerance
        L_w = self.spec.lock_width
        L_d = self.spec.lock_depth
        
        # 2D路径点计算 (简化模型: 矩形+侧面凸凹)
        # 这里为了演示清晰，绘制的是带卡扣的轮廓
        
        path_d = f"M {x_offset} {y_offset} " # Start
        
        # 绘制带互锁的轮廓 (模拟华容道滑块)
        # 底边
        path_d += f"h {S} "
        # 右边 (带凸出锁扣)
        path_d += f"v {S/2 - L_w/2} h {L_d} v {L_w} h {-L_d} v {S/2 - L_w/2} "
        # 顶边
        path_d += f"h {-S} "
        # 左边 (带凹陷锁槽) -> 需要配合公差 T
        path_d += f"v {-S/2 + L_w/2} h {-L_d + T} v {-L_w - T*2} h {L_d - T} v {-S/2 + L_w/2} "
        path_d += "Z"
        
        # 中心刀槽 (矩形)
        slot_w = self.spec.blade_thickness + T
        slot_x = x_offset + (S - slot_w) / 2
        slot_y = y_offset
        
        path_d += f"""<rect x="{slot_x}" y="{slot_y}" width="{slot_w}" height="{S}" 
                      fill="none" stroke="red" stroke-width="0.2" stroke-dasharray="1,1"/>"""
        
        return f'<path d="{path_d}" stroke="black" stroke-width="0.3" fill="none"/>'

    def generate_test_pattern(self) -> str:
        """
        生成测试阵列: 10种模块样本
        目的: 验证互锁强度和打印精度
        """
        svg = [self._generate_header()]
        
        # 绘制 5x2 网格样本
        for row in range(2):
            for col in range(5):
                # 模拟不同磨损程度的公差测试
                # 样本0: 标准公差, 样本4: 宽松公差
                self.spec.tolerance = 0.10 + (col * 0.05)
                
                x = col * (self.spec.nominal_size + 5)
                y = row * (self.spec.nominal_size + 5)
                
                svg.append(self._generate_module_path(x, y))
                
                # 标注
                svg.append(f'<text x="{x}" y="{y-2}" font-size="2">T+{col}</text>')
        
        svg.append("</g></svg>")
        return "".join(svg)

    def save_svg(self, filename: str = "huarong_die_layout.svg"):
        try:
            content = self.generate_test_pattern()
            with open(filename, "w", encoding="utf-8") as f:
                f.write(content)
            print(f"[SUCCESS] 2D Layout generated: {filename}")
            return True
        except Exception as e:
            print(f"[ERROR] Failed to generate layout: {e}")
            return False

- **6mm模块**：刀模行业最小刀间距通用值，作为基础网格单元
- **IADD规格**：刀高、刀厚、桥位、排刀间距等行业标准约束

## 3. 初步分析结论
此问题的本质是**在固定网格(6mm)内设计一套自锁拼接系统**。活字印刷的"铅块独立铸造→排列使用→回收复用"模式可直接映射，但华容道式拼接引入了**平面内互锁**需求，这是与传统活字（仅靠字架外框固定）的关键差异。结构设计的核心矛盾是：**模块越小越灵活，但连接强度和定位精度越难保证**。

class MaterialType(Enum):

@dataclass
class DieModuleSpec:
    """
    刀模单体模块规格 (基于6mm网格)
    Core Logic: 
    1. 基础尺寸 6x6mm (Industry Standard)
    2. 互锁结构: 十字滑块 实现华容道式拼接
    3. 刀槽: 居中
    """
    module_id: str
    grid_unit: float = 6.0       # IADD通用最小网格
    height: float = 23.8         # 标准刀高 (未含底座)
    base_thickness: float = 5.0  # 底座厚度
    
    # 互锁参数
    tolerance: float = 0.15      # 打印公差
    lock_depth: float = 0.8      # 卡扣深度
    lock_width: float = 1.0      # 卡扣宽度
    
    # 刀片参数
    blade_thickness: float = 0.45 # 标准刀厚
    blade_width: float = 0.71    # 重型刀厚
    
    material: MaterialType = MaterialType.PLA

    def __post_init__(self):
        # 根据材料收缩率调整设计尺寸
        shrinkage = self.material.value['shrinkage']
        self.nominal_size = self.grid_unit * (1 + shrinkage)

class DieModuleGenerator:
    """
    生成模块化刀模的2D图纸 (SVG)
    解决问题: 
    - 6mm尺寸下的连接强度
    - PLA高温形变的公差补偿
    """
    
    def __init__(self, spec: DieModuleSpec):
        self.spec = spec
        self.svg_content = []
        
    def _generate_header(self) -> str:
        # 拓竹P2S 打印尺寸 256x256
        return f"""<svg width="256mm" height="256mm" viewBox="0 0 256 256" 
                xmlns="http://www.w3.org/2000/svg">
                <rect width="100%" height="100%" fill="#f0f0f0"/>
                <text x="10" y="20" font-family="Arial" font-size="5" fill="black">
                [P_Huarong] Module Layout | Date: {datetime.datetime.now().strftime('%Y-%m-%d')}
                </text>
                <text x="10" y="28" font-family="Arial" font-size="4" fill="gray">
                Material: {self.spec.material.value['name']} | Tolerance: {self.spec.tolerance}mm
                </text>
                <g transform="translate(20, 40)">
                """

    def _generate_module_path(self, x_offset: float, y_offset: float) -> str:
        """
        生成单个模块的2D投影路径
        结构: 
        - 外轮廓 (6x6mm)
        - 侧面锁扣 (Left/Right Slide, Top/Bottom Slide)
        - 中心刀槽
        """
        S = self.spec.nominal_size
        T = self.spec.tolerance
        L_w = self.spec.lock_width
        L_d = self.spec.lock_depth
        
        # 2D路径点计算 (简化模型: 矩形+侧面凸凹)
        # 这里为了演示清晰，绘制的是带卡扣的轮廓
        
        path_d = f"M {x_offset} {y_offset} " # Start
        
        # 绘制带互锁的轮廓 (模拟华容道滑块)
        # 底边
        path_d += f"h {S} "
        # 右边 (带凸出锁扣)
        path_d += f"v {S/2 - L_w/2} h {L_d} v {L_w} h {-L_d} v {S/2 - L_w/2} "
        # 顶边
        path_d += f"h {-S} "
        # 左边 (带凹陷锁槽) -> 需要配合公差 T
        path_d += f"v {-S/2 + L_w/2} h {-L_d + T} v {-L_w - T*2} h {L_d - T} v {-S/2 + L_w/2} "
        path_d += "Z"
        
        # 中心刀槽 (矩形)
        slot_w = self.spec.blade_thickness + T
        slot_x = x_offset + (S - slot_w) / 2
        slot_y = y_offset
        
        path_d += f"""<rect x="{slot_x}" y="{slot_y}" width="{slot_w}" height="{S}" 
                      fill="none" stroke="red" stroke-width="0.2" stroke-dasharray="1,1"/>"""
        
        return f'<path d="{path_d}" stroke="black" stroke-width="0.3" fill="none"/>'

    def generate_test_pattern(self) -> str:

        svg = [self._generate_header()]
