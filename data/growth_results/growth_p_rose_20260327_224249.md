class RevenueCalculator:
    """
    收益计算器
    基于 L1 数学规律构建收入模型
    
    核心公式:
    R = Σ(付费点_i × 转化率_i × 用户量_i) - 合规成本 - 运营成本
    """
    
    def __init__(self, constraints: KnownConstraints):
        self.constraints = constraints
        self.pricing_tiers = self._init_pricing_tiers()
    
    def _init_pricing_tiers(self) -> List[PricingTier]:
        """
        初始化定价层级
        基于"见路不走"原则：不照搬竞品，从用户价值出发

                price=1.0,  # 1元验证付费意愿
                features=[
                    "高级反馈分析",
                    "数据云同步",
                    "基础报表"
                ],
                target_conversion=0.05  # 目标5%转化
            ),
            PricingTier(
                name="专业版",
                price=29.0,
                features=[
                    "无限任务",
                    "团队协作",
                    "数据导出",
                    "优先支持"
                ],
                target_conversion=0.02  # 目标2%转化
            ),
            PricingTier(
                name="企业版",
                price=299.0,
                features=[
                    "私有部署",
                    "API接入",
                    "定制开发",
                    "专属客服"
                ],
                target_conversion=0.005  # 目标0.5%转化
            ),
        ]
    
    def calculate_monthly_revenue(
        self,
        mau: int,
        conversion_rates: Optional[Dict[str, float]] = None
    ) -> Dict[str, Any]:
        """
        计算月度收入
        
        Args:
            mau: 月活跃用户数
            conversion_rates: 各层级实际转化率 (可选，用于测试)
        
        Returns:
            收入明细和预测
        """
        total_revenue = 0.0
        tier_details = []
        
        for tier in self.pricing_tiers:
            # 使用实际转化率或目标转化率
            conv_rate = (conversion_rates.get(tier.name, tier.target_conversion)
                        if conversion_rates else tier.target_conversion)
            
            users = int(mau * conv_rate)
            revenue = tier.price * users
            
            tier_details.append({
                "name": tier.name,
                "price": tier.price,
                "users": users,
                "conversion": conv_rate,
                "revenue": revenue
            })
            
            total_revenue += revenue
        
        # 计算成本
        compliance_cost = self.constraints.get_total_compliance_cost() / 12  # 分摊到月
        operational_cost = self.constraints.server_cost_monthly
        payment_cost = total_revenue * self.constraints.payment_gateway_fee
        
        net_revenue = total_revenue - operational_cost - payment_cost - compliance_cost
        
        return {
            "gross_revenue": round(total_revenue, 2),
            "compliance_cost": round(compliance_cost, 2),
            "operational_cost": round(operational_cost, 2),
            "payment_cost": round(payment_cost, 2),
            "net_revenue": round(net_revenue, 2),
            "tier_breakdown": tier_details,
            "arpu": round(total_revenue / mau, 4) if mau > 0 else 0
        }
    
    def calculate_break_even_point(
        self,
        monthly_fixed_cost: float,
        average_price: float,
        conversion_rate: float
    ) -> Dict[str, Any]:
        """
        计算盈亏平衡点
        
        盈亏平衡用户数 = 固定成本 / (平均客单价 × 转化率)
        """
        if average_price * conversion_rate == 0:
            return {"error": "价格或转化率不能为0"}
        
        breakeven_users = monthly_fixed_cost / (average_price * conversion_rate)
        breakeven_mau = breakeven_users / conversion_rate
        
        return {
            "monthly_fixed_cost": monthly_fixed_cost,
            "average_price": average_price,
            "conversion_rate": conversion_rate,
            "breakeven_paying_users": math.ceil(breakeven_users),
            "breakeven_mau": math.ceil(breakeven_mau),
            "time_to_breakeven_estimate": f"{math.ceil(breakeven_mau / 100)} 个月 (假设月增100用户)"
        }
    
    def simulate_revenue_scenarios(
        self,
        mau_range: Tuple[int, int, int] = (100, 10000, 100)
    ) -> List[Dict[str, Any]]:
        """
        模拟不同用户规模下的收入场景
        """
        results = []
        start, end, step = mau_range
        
        for mau in range(start, end + 1, step):
            revenue_data = self.calculate_monthly_revenue(mau)
            results.append({
                "mau": mau,
                **revenue_data
            })
        
        return results

    """
    分成方案计算器
    
    处理创作者/平台/推广者的分成比例
    以及税务代扣代缴计算
    """
    
    def __init__(self, constraints: KnownConstraints):
        self.constraints = constraints
    
    def calculate_creator_payout(
        self,
        gross_amount: float,
        share_scheme: RevenueShare
    ) -> Dict[str, Any]:
        """
        计算创作者实际到手金额
        
        规则:
        - 800元以下: 免税
        - 800-4000元: 扣除800后按20%
        - 4000元以上: 扣除20%后按20%
        """
        # 计算分成前金额
        creator_gross = gross_amount * share_scheme.creator_share
        platform_share = gross_amount * share_scheme.platform_share
        promoter_share = gross_amount * share_scheme.promoter_share
        
        # 计算代扣税
        tax = self._calculate_withholding_tax(creator_gross)
        
        # 创作者实际到手
        creator_net = creator_gross - tax
        
        return {
            "gross_transaction": gross_amount,
            "creator_gross": round(creator_gross, 2),
            "creator_tax": round(tax, 2),
            "creator_net": round(creator_net, 2),
            "platform_share": round(platform_share, 2),
            "promoter_share": round(promoter_share, 2),
            "tax_threshold_applied": creator_gross > self.constraints.tax_threshold
        }
    
    def _calculate_withholding_tax(self, amount: float) -> float:
        """
        计算代扣代缴税额
        
        劳务报酬所得预扣预缴规则:
        - 不超过4000元: (收入-800) × 20%
        - 超过4000元: 收入 × (1-20%) × 20%
        """
        if amount <= self.constraints.tax_threshold:
            return 0.0
        
        if amount <= 4000:
            # (收入-800) × 20%
            return (amount - 800) * 0.20

class RevenueCalculator:
    """
    收益计算器
    基于 L1 数学规律构建收入模型
    
    核心公式:
    R = Σ(付费点_i × 转化率_i × 用户量_i) - 合规成本 - 运营成本
    """
    
    def __init__(self, constraints: KnownConstraints):
        self.constraints = constraints
        self.pricing_tiers = self._init_pricing_tiers()
    
    def _init_pricing_tiers(self) -> List[PricingTier]:
        """
        初始化定价层级
        基于"见路不走"原则：不照搬竞品，从用户价值出发

                price=299.0,
                features=[
                    "私有部署",
                    "API接入",
                    "定制开发",
                    "专属客服"
                ],
                target_conversion=0.005  # 目标0.5%转化
            ),
        ]
    
    def calculate_monthly_revenue(
        self,
        mau: int,
        conversion_rates: Optional[Dict[str, float]] = None
    ) -> Dict[str, Any]:
        """
        计算月度收入
        
        Args:
            mau: 月活跃用户数
            conversion_rates: 各层级实际转化率 (可选，用于测试)
        
        Returns:
            收入明细和预测
        """
        total_revenue = 0.0
        tier_details = []
        
        for tier in self.pricing_tiers:
            # 使用实际转化率或目标转化率
            conv_rate = (conversion_rates.get(tier.name, tier.target_conversion)
                        if conversion_rates else tier.target_conversion)
            
            users = int(mau * conv_rate)
            revenue = tier.price * users
            
            tier_details.append({
                "name": tier.name,
                "price": tier.price,
                "users": users,
                "conversion": conv_rate,
                "revenue": revenue
            })
            
            total_revenue += revenue
        
        # 计算成本
        compliance_cost = self.constraints.get_total_compliance_cost() / 12  # 分摊到月
        operational_cost = self.constraints.server_cost_monthly
        payment_cost = total_revenue * self.constraints.payment_gateway_fee
        
        net_revenue = total_revenue - operational_cost - payment_cost - compliance_cost
        
        return {
            "gross_revenue": round(total_revenue, 2),
            "compliance_cost": round(compliance_cost, 2),
            "operational_cost": round(operational_cost, 2),
            "payment_cost": round(payment_cost, 2),
            "net_revenue": round(net_revenue, 2),
            "tier_breakdown": tier_details,
            "arpu": round(total_revenue / mau, 4) if mau > 0 else 0
        }
    
    def calculate_break_even_point(
        self,
        monthly_fixed_cost: float,
        average_price: float,
        conversion_rate: float
    ) -> Dict[str, Any]:
        """
        计算盈亏平衡点
        
        盈亏平衡用户数 = 固定成本 / (平均客单价 × 转化率)
        """
        if average_price * conversion_rate == 0:
            return {"error": "价格或转化率不能为0"}
        
        breakeven_users = monthly_fixed_cost / (average_price * conversion_rate)
        breakeven_mau = breakeven_users / conversion_rate
        
        return {
            "monthly_fixed_cost": monthly_fixed_cost,
            "average_price": average_price,
            "conversion_rate": conversion_rate,
            "breakeven_paying_users": math.ceil(breakeven_users),
            "breakeven_mau": math.ceil(breakeven_mau),
            "time_to_breakeven_estimate": f"{math.ceil(breakeven_mau / 100)} 个月 (假设月增100用户)"
        }
    
    def simulate_revenue_scenarios(
        self,
        mau_range: Tuple[int, int, int] = (100, 10000, 100)
    ) -> List[Dict[str, Any]]:
        """
        模拟不同用户规模下的收入场景
        """
        results = []
        start, end, step = mau_range
        
        for mau in range(start, end + 1, step):
            revenue_data = self.calculate_monthly_revenue(mau)
            results.append({
                "mau": mau,
                **revenue_data
            })
        
        return results

    def calculate_creator_payout(
        self,

        share_scheme: RevenueShare
    ) -> Dict[str, Any]:
        """

        - 800元以下: 免税
        - 800-4000元: 扣除800后按20%
        - 4000元以上: 扣除20%后按20%

        creator_gross = gross_amount * share_scheme.creator_share
        platform_share = gross_amount * share_scheme.platform_share
        promoter_share = gross_amount * share_scheme.promoter_share
        
        # 计算代扣税
        tax = self._calculate_withholding_tax(creator_gross)
        
        # 创作者实际到手
        creator_net = creator_gross - tax
        
        return {
            "gross_transaction": gross_amount,
            "creator_gross": round(creator_gross, 2),
            "creator_tax": round(tax, 2),
            "creator_net": round(creator_net, 2),
            "platform_share": round(platform_share, 2),
            "promoter_share": round(promoter_share, 2),
            "tax_threshold_applied": creator_gross > self.constraints.tax_threshold
        }

                    "API接入",
                    "定制开发",
                    "专属客服"
                ],
                target_conversion=0.005  # 目标0.5%转化
            ),
        ]
    
    def calculate_monthly_revenue(
        self,
        mau: int,
        conversion_rates: Optional[Dict[str, float]] = None
    ) -> Dict[str, Any]:
        """
        计算月度收入
        
        Args:
            mau: 月活跃用户数
            conversion_rates: 各层级实际转化率 (可选，用于测试)
        
        Returns:
            收入明细和预测
        """
        total_revenue = 0.0
        tier_details = []
        
        for tier in self.pricing_tiers:
            # 使用实际转化率或目标转化率
            conv_rate = (conversion_rates.get(tier.name, tier.target_conversion)
                        if conversion_rates else tier.target_conversion)
            
            users = int(mau * conv_rate)
            revenue = tier.price * users
            
            tier_details.append({
                "name": tier.name,
                "price": tier.price,
                "users": users,
                "conversion": conv_rate,
                "revenue": revenue
            })
            
            total_revenue += revenue
        
        # 计算成本
        compliance_cost = self.constraints.get_total_compliance_cost() / 12  # 分摊到月
        operational_cost = self.constraints.server_cost_monthly
        payment_cost = total_revenue * self.constraints.payment_gateway_fee
        
        net_revenue = total_revenue - operational_cost - payment_cost - compliance_cost
        
        return {
            "gross_revenue": round(total_revenue, 2),
            "compliance_cost": round(compliance_cost, 2),
            "operational_cost": round(operational_cost, 2),
            "payment_cost": round(payment_cost, 2),
            "net_revenue": round(net_revenue, 2),
            "tier_breakdown": tier_details,
            "arpu": round(total_revenue / mau, 4) if mau > 0 else 0
        }
    
    def calculate_break_even_point(
        self,
        monthly_fixed_cost: float,
        average_price: float,
        conversion_rate: float
    ) -> Dict[str, Any]:
        """
        计算盈亏平衡点
        
        盈亏平衡用户数 = 固定成本 / (平均客单价 × 转化率)
        """
        if average_price * conversion_rate == 0:
            return {"error": "价格或转化率不能为0"}
        
        breakeven_users = monthly_fixed_cost / (average_price * conversion_rate)
        breakeven_mau = breakeven_users / conversion_rate
        
        return {
            "monthly_fixed_cost": monthly_fixed_cost,
            "average_price": average_price,
            "conversion_rate": conversion_rate,
            "breakeven_paying_users": math.ceil(breakeven_users),
            "breakeven_mau": math.ceil(breakeven_mau),
            "time_to_breakeven_estimate": f"{math.ceil(breakeven_mau / 100)} 个月 (假设月增100用户)"
        }
    
    def simulate_revenue_scenarios(
        self,
        mau_range: Tuple[int, int, int] = (100, 10000, 100)
    ) -> List[Dict[str, Any]]:
        """
        模拟不同用户规模下的收入场景
        """
        results = []
        start, end, step = mau_range
        
        for mau in range(start, end + 1, step):
            revenue_data = self.calculate_monthly_revenue(mau)
            results.append({
                "mau": mau,
                **revenue_data
            })
        
        return results

                    "专属客服"
                ],
                target_conversion=0.005  # 目标0.5%转化
            ),
        ]
    
    def calculate_monthly_revenue(
        self,
        mau: int,
        conversion_rates: Optional[Dict[str, float]] = None
    ) -> Dict[str, Any]:
        """
        计算月度收入
        
        Args:
            mau: 月活跃用户数
            conversion_rates: 各层级实际转化率 (可选，用于测试)
        
        Returns:
            收入明细和预测
        """
        total_revenue = 0.0
        tier_details = []
        
        for tier in self.pricing_tiers:
            # 使用实际转化率或目标转化率
            conv_rate = (conversion_rates.get(tier.name, tier.target_conversion)
                        if conversion_rates else tier.target_conversion)
            
            users = int(mau * conv_rate)
            revenue = tier.price * users
            
            tier_details.append({
                "name": tier.name,
                "price": tier.price,
                "users": users,
                "conversion": conv_rate,
                "revenue": revenue
            })
            
            total_revenue += revenue
        
        # 计算成本
        compliance_cost = self.constraints.get_total_compliance_cost() / 12  # 分摊到月
        operational_cost = self.constraints.server_cost_monthly
        payment_cost = total_revenue * self.constraints.payment_gateway_fee
        
        net_revenue = total_revenue - operational_cost - payment_cost - compliance_cost
        
        return {
            "gross_revenue": round(total_revenue, 2),
            "compliance_cost": round(compliance_cost, 2),
            "operational_cost": round(operational_cost, 2),
            "payment_cost": round(payment_cost, 2),
            "net_revenue": round(net_revenue, 2),
            "tier_breakdown": tier_details,
            "arpu": round(total_revenue / mau, 4) if mau > 0 else 0
        }
    
    def calculate_break_even_point(
        self,
        monthly_fixed_cost: float,
        average_price: float,
        conversion_rate: float
    ) -> Dict[str, Any]:
        """
        计算盈亏平衡点
        
        盈亏平衡用户数 = 固定成本 / (平均客单价 × 转化率)
        """
        if average_price * conversion_rate == 0:
            return {"error": "价格或转化率不能为0"}
        
        breakeven_users = monthly_fixed_cost / (average_price * conversion_rate)
        breakeven_mau = breakeven_users / conversion_rate
        
        return {
            "monthly_fixed_cost": monthly_fixed_cost,
            "average_price": average_price,
            "conversion_rate": conversion_rate,
            "breakeven_paying_users": math.ceil(breakeven_users),
            "breakeven_mau": math.ceil(breakeven_mau),
            "time_to_breakeven_estimate": f"{math.ceil(breakeven_mau / 100)} 个月 (假设月增100用户)"
        }
    
    def simulate_revenue_scenarios(
        self,
        mau_range: Tuple[int, int, int] = (100, 10000, 100)
    ) -> List[Dict[str, Any]]:
        """
        模拟不同用户规模下的收入场景
        """
        results = []
        start, end, step = mau_range
        
        for mau in range(start, end + 1, step):
            revenue_data = self.calculate_monthly_revenue(mau)
            results.append({
                "mau": mau,
                **revenue_data
            })
        
        return results

        total_revenue = 0.0
        tier_details = []
        
        for tier in self.pricing_tiers:
            # 使用实际转化率或目标转化率
            conv_rate = (conversion_rates.get(tier.name, tier.target_conversion)
                        if conversion_rates else tier.target_conversion)
            
            users = int(mau * conv_rate)
            revenue = tier.price * users
            
            tier_details.append({
                "name": tier.name,
                "price": tier.price,
                "users": users,
                "conversion": conv_rate,
                "revenue": revenue
            })
            
            total_revenue += revenue
        
        # 计算成本
        compliance_cost = self.constraints.get_total_compliance_cost() / 12  # 分摊到月
        operational_cost = self.constraints.server_cost_monthly
        payment_cost = total_revenue * self.constraints.payment_gateway_fee
        
        net_revenue = total_revenue - operational_cost - payment_cost - compliance_cost
        
        return {
            "gross_revenue": round(total_revenue, 2),
            "compliance_cost": round(compliance_cost, 2),
            "operational_cost": round(operational_cost, 2),
            "payment_cost": round(payment_cost, 2),
            "net_revenue": round(net_revenue, 2),
            "tier_breakdown": tier_details,
            "arpu": round(total_revenue / mau, 4) if mau > 0 else 0
        }
    
    def calculate_break_even_point(
        self,
        monthly_fixed_cost: float,
        average_price: float,
        conversion_rate: float
    ) -> Dict[str, Any]:
        """
        计算盈亏平衡点
        
        盈亏平衡用户数 = 固定成本 / (平均客单价 × 转化率)
        """
        if average_price * conversion_rate == 0:
            return {"error": "价格或转化率不能为0"}
        
        breakeven_users = monthly_fixed_cost / (average_price * conversion_rate)
        breakeven_mau = breakeven_users / conversion_rate
        
        return {
            "monthly_fixed_cost": monthly_fixed_cost,
            "average_price": average_price,
            "conversion_rate": conversion_rate,
            "breakeven_paying_users": math.ceil(breakeven_users),
            "breakeven_mau": math.ceil(breakeven_mau),
            "time_to_breakeven_estimate": f"{math.ceil(breakeven_mau / 100)} 个月 (假设月增100用户)"
        }
    
    def simulate_revenue_scenarios(
        self,
        mau_range: Tuple[int, int, int] = (100, 10000, 100)
    ) -> List[Dict[str, Any]]:
        """
        模拟不同用户规模下的收入场景
        """
        results = []
        start, end, step = mau_range
        
        for mau in range(start, end + 1, step):
            revenue_data = self.calculate_monthly_revenue(mau)
            results.append({
                "mau": mau,
                **revenue_data
            })
        
        return results
