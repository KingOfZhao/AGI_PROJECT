**关键技术**: Fabric.js Canvas → THREE.CanvasTexture

```javascript
// DesignTextureManager.js
class DesignTextureManager {
  constructor(threeScene) {
    this.textures = new Map() // panelId → THREE.CanvasTexture
  }
  
  updatePanelTexture(panelId, fabricCanvas) {
    const canvasEl = fabricCanvas.getElement()
    
    if (this.textures.has(panelId)) {
      // 更新已有贴图
      this.textures.get(panelId).needsUpdate = true
    } else {
      // 创建新贴图
      const tex = new THREE.CanvasTexture(canvasEl)
      tex.colorSpace = THREE.SRGBColorSpace
      tex.flipY = false
      this.textures.set(panelId, tex)
    }
    
    // 找到对应的3D面板mesh, 设置贴图
    const mesh = this.findPanelMesh(panelId)
    if (mesh) {
      mesh.material.map = this.textures.get(panelId)
      mesh.material.needsUpdate = true
    }
  }
}
```

```javascript
function makePanelFromShape(shapePoints, thickness, material) {
  const shape = new THREE.Shape()
  const pts = shapePoints
  
  shape.moveTo(pts[0].x, pts[0].y)
  for (let i = 1; i < pts.length; i++) {
    const p = pts[i]
    if (p.type === 'arc') {
      shape.absellipse(p.cx, p.cy, p.rx, p.ry, p.startAngle, p.endAngle, p.ccw)
    } else if (p.type === 'bezier') {
      shape.bezierCurveTo(p.cp1x, p.cp1y, p.cp2x, p.cp2y, p.x, p.y)
    } else {
      shape.lineTo(p.x, p.y)
    }
  }
  shape.closePath()
  
  const geometry = new THREE.ExtrudeGeometry(shape, {
    depth: thickness,
    bevelEnabled: false,
  })
  
  // 材质: [外面, 内面, 侧面]
  const mesh = new THREE.Mesh(geometry, [
    material.outside,
    material.inside, 
    material.outside,
  ])
  mesh.castShadow = true
  mesh.receiveShadow = true
  return mesh
}
```

```typescript
interface FoldNode {
  id: string              // 'B', 'BL', 'BL1', 'FL1'...
  shape: ShapePoint[]     // 2D轮廓点
  position: [number, number, number]  // 相对于父铰链的位置
  children: FoldEdge[]    // 子节点(通过铰链连接)
}

interface FoldEdge {
  id: string              // 'B_BL', 'BL_BL1'...
  axis: 'x' | 'y' | 'z'  // 铰链旋转轴
  hingePosition: [number, number, number]  // 铰链位置
  hingeLength: number     // 铰链长度(mm)
  foldAngle: number       // 目标折叠角度(弧度)
  springback: number      // 回弹量(弧度)
  child: FoldNode         // 子面板
}
```

### 碰撞方向2: 递归构建 Three.js Group

```javascript
function buildFoldTree(node, parentGroup, materials) {
  // 1. 创建面板mesh
  const mesh = makePanelFromShape(node.shape, thickness, materials)
  mesh.name = node.id
  
  // 2. 对每个子边(铰链), 创建Group
  for (const edge of node.children) {
    const hingeGroup = new THREE.Group()
    hingeGroup.name = edge.id
    hingeGroup.position.set(...edge.hingePosition)
    
    // 3. 铰链圆柱可视化
    const hingeMesh = makeHingeCylinder(edge.hingeLength, 0.26)
    hingeGroup.add(hingeMesh)
    
    // 4. 递归构建子树
    buildFoldTree(edge.child, hingeGroup, materials)
    
    mesh.add(hingeGroup)
  }
  
  parentGroup.add(mesh)
}
```

### 碰撞方向3: 折叠动画(GSAP)

```javascript
import gsap from 'gsap'

function animateFold(foldTree, targetAngle, duration = 1.5) {
  const tl = gsap.timeline()
  
  // 遍历所有铰链, 按层级顺序添加动画
  function addAnimations(node, depth) {
    for (const edge of node.children) {
      const group = scene.getObjectByName(edge.id)
      if (!group) continue
      
      const axisKey = `rotation.${edge.axis}`
      const angle = edge.foldAngle * targetAngle // targetAngle: 0→1 (展开→折叠)
      
      tl.to(group.rotation, {
        [edge.axis]: angle,
        duration: duration / (depth + 1),
        ease: 'power2.inOut',
      }, depth * 0.15) // 按层级延迟
      
      addAnimations(edge.child, depth + 1)
    }
  }
  
  addAnimations(foldTree, 0)
  return tl
}
```

```javascript
function generate_120370(L, W, H, t) {
  const earW = Math.min(20, H * 0.8) // 折耳宽度
  const tabW = Math.min(15, H * 0.6) // 粘口宽度
  
  // 2D布局坐标系 (原点=底板左下角)
  const panels = {
    B:  { x: H+earW, y: H+earW, w: L, h: W },           // 底板
    F:  { x: H+earW, y: H+earW-H, w: L, h: H },         // 前壁
    H:  { x: H+earW, y: H+earW+W, w: L, h: H },         // 后壁
    BL: { x: earW, y: H+earW, w: H, h: W },              // 左壁
    BR: { x: H+earW+L, y: H+earW, w: H, h: W },         // 右壁
  }
  
  // 四角45°三角形
  const corners = [
    { id: 'FL', vertices: [[panels.B.x, panels.F.y+H], [panels.BL.x+H, panels.F.y+H], [panels.BL.x+H, panels.F.y]] },
    { id: 'FR', vertices: [[panels.B.x+L, panels.F.y+H], [panels.BR.x, panels.F.y+H], [panels.BR.x, panels.F.y]] },
    { id: 'HL', vertices: [[panels.B.x, panels.H.y], [panels.BL.x+H, panels.H.y], [panels.BL.x+H, panels.H.y+H]] },
    { id: 'HR', vertices: [[panels.B.x+L, panels.H.y], [panels.BR.x, panels.H.y], [panels.BR.x, panels.H.y+H]] },
  ]
  
  // 折耳(梯形)
  const ears = generateTrapezoidEars(panels, earW, tabW)
  
  // 生成2D实体
  const entities = []
  // 切割线
  for (const [id, p] of Object.entries(panels)) {
    entities.push(...rectToLines(p, 'cut'))
  }
  // 折痕线
  entities.push(line(panels.B.x, panels.B.y, panels.B.x+L, panels.B.y, 'crease'))
  entities.push(line(panels.B.x, panels.B.y+W, panels.B.x+L, panels.B.y+W, 'crease'))
  entities.push(line(panels.B.x, panels.B.y, panels.B.x, panels.B.y+W, 'crease'))
  entities.push(line(panels.B.x+L, panels.B.y, panels.B.x+L, panels.B.y+W, 'crease'))
  // 斜切线
  for (const c of corners) {
    entities.push(triangleToLines(c, 'cut'))
  }
  
  // 尺寸标注
  const annotations = [
    { x: panels.B.x + L/2, y: panels.F.y - 15, label: 'L', value: L, unit: 'mm' },
    { x: panels.BR.x + H + 15, y: panels.B.y + W/2, label: 'W', value: W, unit: 'mm' },
    { x: panels.B.x + L/2, y: panels.H.y + H + 15, label: 'H', value: H, unit: 'mm' },
  ]
  
  const totalX = earW + H + L + H + earW
  const totalY = earW + H + W + H + earW
  
  return { panels, corners, ears, entities, annotations, bounds: { totalX, totalY } }
}
```

```javascript
function applyMaterial(scene, presetId) {
  const preset = MATERIAL_PRESETS[presetId]
  const mat = new THREE.MeshPhysicalMaterial({
    color: preset.color,
    roughness: preset.roughness,
    metalness: preset.metalness,
    clearcoat: preset.clearcoat,
    clearcoatRoughness: 0.1,
    side: THREE.FrontSide,
  })
  
  // 加载纹理
  if (preset.textureUrl) {
    new THREE.TextureLoader().load(preset.textureUrl, (tex) => {
      tex.colorSpace = THREE.SRGBColorSpace
      mat.map = tex
      mat.needsUpdate = true
    })
  }
  
  // 应用到所有面板
  scene.traverse((child) => {
    if (child.isMesh && child.name && !child.name.includes('_')) {
      // 面板mesh (不含铰链)
      child.material = [mat, mat.clone(), mat] // outside/inside/side
    }
  })
}
```
